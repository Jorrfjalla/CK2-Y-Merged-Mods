targetted_decisions = {
	mp_alliance = { #Create alliance
		from_potential = { #Player only
			is_alive = yes
			ai = no
			NOT = {
				has_character_flag = hidden_mp_commands
			}
		}
		potential = { #Targeted to another player only, and not already in an alliance with each other
			is_alive = yes
			ai = no
			has_game_rule = {
				name = multiplayer_commands
				value = enabled
			}
			NOT = {
				AND = {
					ROOT = {
						has_character_flag = mp_alliance
					}
					FROM = {
						has_character_flag = mp_alliance
					}
				}
			}
		}
		allow = { #2nd check to make sure not in alliance together already
			NOT = {
				AND = {
					ROOT = {
						has_character_flag = mp_alliance
					}
					FROM = {
						has_character_flag = mp_alliance
					}
				}
			}
		}
		effect = { #Adds mp_alliance flag to both parties, also creates 1000 year alliance, this is overkill, but with the flag system, each individual character can only do this once
			FROM = { 
				add_alliance = { 
				who = ROOT 
				years = 1000 
				}
				set_character_flag = mp_alliance
			}
			ROOT = {
				add_alliance = {
				who = FROM 
				years = 1000 
				}
				set_character_flag = mp_alliance
			}
		}
		revoke_allowed = { #Unsure what this does, since you can still break the alliance manually
			always = no
		}
	}

	mp_friend = { #Add mutual friendsip
		from_potential = { #Player only
			is_alive = yes
			ai = no
			NOT = { is_friend = FROM }
			NOT = {
				has_character_flag = hidden_mp_commands
			}
		}
		potential = { #Targeted to player only, who is not already a friend
			is_alive = yes
			ai = no
			has_game_rule = {
				name = multiplayer_commands
				value = enabled
			}
			NOT = { is_friend = FROM }
		}
		allow = { #Another check if friends or not
			NOT = { is_friend = FROM }
		}
		effect = { #Add friend status
			FROM = { 
				add_friend = ROOT
			}
		}
		revoke_allowed = { #Unsure what this does
			always = no
		}
	}

				name = multiplayer_commands
				value = enabled
			}
			FROM = {
				wealth >= 100
			}
		}
		allow = { #Additional check to ensure enough gold
			FROM = {
				wealth >= 100
			}
		}
		effect = { #Give 100 gold to target, take away 100 gold from source
			FROM = { wealth = -100 }
			ROOT = { wealth = 100 }
		}
		revoke_allowed = {
			always = no
		}
	}
