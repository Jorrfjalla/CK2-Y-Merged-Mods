targetted_decisions = {
	mp_alliance = { #Create alliance
		ai = no #Stops ai even considering this, performance tweak
		from_potential = { #Player only
			is_alive = yes
			ai = no
			NOT = {
				has_character_flag = hidden_mp_commands
			}
		}
		potential = { #Targeted to another player only, and not already in an alliance with each other
			is_alive = yes
			ai = no
			has_game_rule = {
				name = multiplayer_commands
				value = enabled
			}
			NOT = {
				ROOT = {
					is_allied_with = FROM
				}
			}
		}
		allow = { #2nd check to make sure not in alliance together already
			NOT = {
				ROOT = {
					is_allied_with = FROM
				}
			}
		}
		effect = { #Adds mp_alliance flag to both parties, also creates 1000 year alliance, this is overkill
			FROM = { 
				add_alliance = { 
					who = ROOT
					years = 1000 
				}
			}
			ROOT = {
				add_alliance = {
					who = FROM 
					years = 1000 
				}
			}
		}
		revoke_allowed = { #Unsure what this does, same with the wiki, it's apparently not used in vanilla
			always = no
		}
	}
