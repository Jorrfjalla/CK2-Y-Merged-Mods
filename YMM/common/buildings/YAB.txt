		desc = ca_checkpoint_1_desc
        potential = { #Hides when not with cheat bloodline, hence no need for gamerule intervention, only for direct owner as well, no liege checking here

				any_owned_bloodline {
					has_bloodline_flag = bloodline_yan
				}
            }
        }
		trigger = { TECH_FORTIFICATIONS_CONSTRUCTION = 2 } #Shows, but blocks construction until tech requirement is met

		build_time = 365 #Time in days

		levy_size = 0.1
		tax_income = 0.5

		ai_creation_factor = 0 #We don't want the AI making this, even if they have the bloodline

	}
	ca_checkpoint_2 = {
		desc = ca_checkpoint_2_desc

        }
